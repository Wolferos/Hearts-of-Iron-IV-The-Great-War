
is_friend_or_potential_friend_of_ENG = {
	OR = {
		is_puppet_of = ENG
		is_puppet = no
	}
	OR = {
		is_in_faction_with = FRA
		is_in_faction = no
	}
	NOT = { has_war_with = ENG }
	has_capitulated = no
	exists = yes
}

is_alone_or_in_faction_at_peace_with_ENG = {
	OR = {
		is_in_faction = no
		is_in_faction_with = FRA
		AND = {
			is_in_faction = yes
			NOT = {
				is_in_faction_with = FRA
			}
		}
	}
	NOT = {
		any_country = {
			is_in_faction_with = PREV
			has_war_with = ENG
		}
	}
}
