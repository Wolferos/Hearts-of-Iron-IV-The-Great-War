# Italy Honour the Pact ?

country_event = {
	id = ww1_italy.6
	title = ww1_italy.6.t
	desc = ww1_italy.6.t
	#picture = ww1_italy_6

	fire_only_once = yes

	trigger = {
		TAG = ITA
		OR = {
			AND = {
				GER = {	has_war_with = FRA }
				GER = {	has_war_with = ENG }
			}
			AND = {
				GER = {	has_war_with = RUS }
				GER = {	has_war_with = ENG }
				GER = {	has_war_with = FRA }
			}
		}
	}
	
	mean_time_to_happen = {
		days = 10
	}

	# Honour
	option = {
		name = ww1_italy.6.a
		ai_chance = {
			factor = 5
			modifier = {
				factor = 80
				AUH = {
					has_completed_focus = AUH_ww1_give_italy_territories
				}
			}
		}
		
		GER = {
			add_to_faction = ITA
		}

		ITA = {
			add_to_war = {
	            targeted_alliance = GER enemy = BEL
	        }
		}


		hidden_effect = {
			news_event = { days = 1 id = ww1_italy.7 }
		}
	}

	# Don't Honour
	option = {
		name = ww1_italy.6.b
		ai_chance = {
			factor = 95
			modifier = {
				factor = 20
				AUH = {
					has_completed_focus = AUH_ww1_give_italy_territories
				}
			}
		}
	}	
}
