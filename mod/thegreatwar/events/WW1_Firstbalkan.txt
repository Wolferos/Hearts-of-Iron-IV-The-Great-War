# Declaration of War - Triggered by SER
#____________________________________________________________
country_event = {
	id = ww1_firstbalkan.4
	title = ww1_firstbalkan.4.t
	desc = ww1_firstbalkan.4.d
	#picture = ww1_firstbalkan_4

	is_triggered_only = yes
	
	# SER Yes
	option = {
		name = ww1_firstbalkan.4.a
		ai_chance = { factor = 100 }
		# Serbia
		SER = {
			declare_war_on = {
			    type = take_state_focus 
			    target = TUR 
			    generator = {
			        106 757 758
			    }
			}
		}

		hidden_effect = {
			SER = {
				add_named_threat = {
					threat = 10
					name = "War in the Balkans"
				}
			}
			set_global_flag = kis_firstbalkan_war
			news_event = { days = 1 id = ww1_firstbalkan.100 }

			44 = { set_border_war = no }

			SER = {
				add_manpower = 15000
			}			
			BUL = {
				add_manpower = 15000
			}
			GRE = {
				add_manpower = 15000
			}
			MTN = {
				add_manpower = 3500
			}
		}

		# Bulgaria
		if = {
			limit = {
				OR = {
					BUL = {
						is_in_faction_with = SER
						is_in_faction_with = GRE
						is_in_faction_with = MTN
					}
				}
			}
			BUL = {
				create_wargoal = {
		            type = take_state_focus 
		            target = TUR 
		            generator = {
		                757 754 755 760
		            }
		        }
		        add_to_war = {
		            targeted_alliance = SER enemy = TUR
		        }
			}
		}

		# Greece
		if = {
			limit = {
				OR = {
					GRE = {
						is_in_faction_with = SER
						is_in_faction_with = BUL
						is_in_faction_with = MTN
					}
				}
			}
			GRE = {
				create_wargoal = {
		            type = take_state_focus 
		            target = TUR 
		            generator = {
		                731 756 184
		            }
		        }
		        add_to_war = {
		            targeted_alliance = SER enemy = TUR
		        }
		    }
		}

		# Montenegro
		if = {
			limit = {
				OR = {
					MTN = {
						is_in_faction_with = SER
						is_in_faction_with = BUL
						is_in_faction_with = GRE
					}
				}
			}
			MTN = {
				create_wargoal = {
		            type = take_state_focus 
		            target = TUR 
		            generator = {
		                759
		            }
		        }
		        add_to_war = {
		            targeted_alliance = SER enemy = TUR
		        }
			}
		}

	}
}
