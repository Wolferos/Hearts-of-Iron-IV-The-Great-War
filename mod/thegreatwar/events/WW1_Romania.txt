	}

	}

					name="Ferdinand I"

					traits={}
				}

			}
		}
	}

	option={
		name = ww1_romania.1.a
		ai_chance={
			factor=100
		}	
	}
}

		}

				modifier=large_increase
			}
		}

				modifier=large_increase
			}
		}

			}

			}

			}

				target=GER

					remove_from_faction=ROM
				}
			}
